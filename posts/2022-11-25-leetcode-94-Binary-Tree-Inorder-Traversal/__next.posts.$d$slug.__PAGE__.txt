  <span class="hljs-keyword">def</span> <span class="hljs-title function_">inorderDfs</span>(<span class="hljs-params">node</span>):
    <span class="hljs-keyword">if</span> <span class="hljs-keyword">not</span> node: <span class="hljs-keyword">return</span>
    <span class="hljs-keyword">if</span> node.left:
      inorderDfs(node.left)
